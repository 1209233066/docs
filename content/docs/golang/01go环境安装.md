### windows环境安装

```shell
GO111MODULE=on
GOPATH=d:\gopath
GOPROXY=https://goproxy.cn
GOROOT=D:\go
```

```shell
go version
go env
```



### 在linux环境安装

```bash
# centos 启动桌面环境
yum groupinstall "X Window System"
yum groupinstall -y "GNOME Desktop"
init 5
```

```bash
# ubuntu server 启动桌面环境
sudo apt install ubuntu-desktop

init 5
```



安装golang
```bash
#https://studygolang.com/dl
wget https://golang.google.cn/dl/go1.19.6.linux-amd64.tar.gz
tar xf go1.19.6.linux-amd64.tar.gz -C /opt/
mkdir /home/pc/golang -p
```

```bash
root@golang:~# ls /opt/go/
api  codereview.cfg   doc  LICENSE  PATENTS  README.md    src   VERSION
bin  CONTRIBUTING.md  lib  misc     pkg      SECURITY.md  test
```



```bash
cat >>/etc/profile <<EOF
export GOROOT=/opt/go
export GOPATH=/opt/gopath
export PATH=\$GOROOT/bin:$PATH
export GO111MODULE="on"
export GOPROXY="https://goproxy.cn"
EOF
```



```bash
source /etc/profile
go version
go env
```

### demo

```bash
mkdir -p lvminfo/{utils,cmd} 
go mod init wangendao.com/lvminfo
```

```bash
lvminfo/
├── cmd
│   └── main.go		# 入口文件
├── go.mod
└── utils			# 内部包存放目录
    └── getinfo.go
```

```go
// utils/getinfo.go
package utils

import "os/exec"

func Getlvinfo() []byte {
	lvs := exec.Command("lvs")
	result, _ := lvs.Output()
	return result
}
```

```go
// cmd/main.go
package main

import (
	"fmt"

	"wangendao.com/lvminfo/utils"
)

func main() {
	a := utils.Getlvinfo()
	fmt.Println(string(a))
}
```
